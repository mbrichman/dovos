    <script>
        document.addEventListener('DOMContentLoaded', function() {
    // Get filter elements
    const sourceFilter = document.getElementById('sourceFilter');
    const dateFilter = document.getElementById('dateFilter');
    const sortOrder = document.getElementById('sortOrder');
    
    // Set up filtering
    setupFilters();
    
    // Apply URL parameters to filters on page load
    applyUrlParamsToFilters();
    
    // Format dates
    formatDates();
    
    // Initial application of filters
    applyFilters();
    
    function setupFilters() {
        if (!sourceFilter || !dateFilter || !sortOrder) return;
        
        const filters = [sourceFilter, dateFilter, sortOrder];
        
        filters.forEach(filter => {
            filter.addEventListener('change', function() {
                applyFilters();
                updateUrlWithFilters();
            });
        });
    }
    
    function applyUrlParamsToFilters() {
        // Get URL parameters
        const urlParams = new URLSearchParams(window.location.search);
        
        // Apply source filter from URL if present
        if (urlParams.has('source') && sourceFilter) {
            const sourceValue = urlParams.get('source');
            sourceFilter.value = sourceValue;
        }
        
        // Apply date filter from URL if present
        if (urlParams.has('date') && dateFilter) {
            const dateValue = urlParams.get('date');
            dateFilter.value = dateValue;
        }
        
        // Apply sort filter from URL if present
        if (urlParams.has('sort') && sortOrder) {
            const sortValue = urlParams.get('sort');
            sortOrder.value = sortValue;
        }
    }
    
    function updateUrlWithFilters() {
        // Create URL parameters object
        const urlParams = new URLSearchParams(window.location.search);
        
        // Set current page to 1 when filters change
        urlParams.set('page', '1');
        
        // Update source parameter
        if (sourceFilter && sourceFilter.value !== 'all') {
            urlParams.set('source', sourceFilter.value);
        } else {
            urlParams.delete('source');
        }
        
        // Update date parameter
        if (dateFilter && dateFilter.value !== 'all') {
            urlParams.set('date', dateFilter.value);
        } else {
            urlParams.delete('date');
        }
        
        // Update sort parameter
        if (sortOrder && sortOrder.value !== 'newest') {
            urlParams.set('sort', sortOrder.value);
        } else {
            urlParams.delete('sort');
        }
        
        // Update URL without reload
        const newUrl = window.location.pathname + (urlParams.toString() ? '?' + urlParams.toString() : '');
        window.history.pushState({}, '', newUrl);
    }
    
    function applyFilters() {
        const sourceValue = sourceFilter ? sourceFilter.value : 'all';
        const dateValue = dateFilter ? dateFilter.value : 'all';
        const sortValue = sortOrder ? sortOrder.value : 'newest';
        
        // Get all conversation cards
        const cards = document.querySelectorAll('.conversation-card');
        
        cards.forEach(card => {
            // Start with visible
            let shouldShow = true;
            
            // Apply source filter
            if (sourceValue !== 'all') {
                const cardSource = card.getAttribute('data-source');
                if (cardSource !== sourceValue) {
                    shouldShow = false;
                }
            }
            
            // Apply date filter
            if (shouldShow && dateValue !== 'all') {
                const cardDate = new Date(card.getAttribute('data-date'));
                const now = new Date();
                
                if (isNaN(cardDate.getTime())) {
                    // If date parsing fails, show the card anyway
                } else {
                    // Apply date filters
                    if (dateValue === 'today') {
                        if (cardDate.toDateString() !== now.toDateString()) {
                            shouldShow = false;
                        }
                    } else if (dateValue === 'week') {
                        const weekAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
                        if (cardDate < weekAgo) {
                            shouldShow = false;
                        }
                    } else if (dateValue === 'month') {
                        const monthAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);
                        if (cardDate < monthAgo) {
                            shouldShow = false;
                        }
                    } else if (dateValue === 'year') {
                        const yearAgo = new Date(now.getTime() - 365 * 24 * 60 * 60 * 1000);
                        if (cardDate < yearAgo) {
                            shouldShow = false;
                        }
                    }
                }
            }
            
            // Show or hide the card
            card.style.display = shouldShow ? 'block' : 'none';
        });
    }
    
    function formatDates() {
        const dateElements = document.querySelectorAll('.date-to-format');
        
        dateElements.forEach(function(element) {
            const dateStr = element.textContent.trim();
            const formattedDate = formatDateNicely(dateStr);
            element.textContent = formattedDate;
        });
    }
    
    // Update pagination links to preserve filter parameters
    updatePaginationLinks();
    
    function updatePaginationLinks() {
        const paginationLinks = document.querySelectorAll('.pagination .page-link');
        
        paginationLinks.forEach(link => {
            const href = link.getAttribute('href');
            if (!href || href === '#' || link.parentElement.classList.contains('disabled')) {
                return;
            }
            
            // Parse current URL to get the page parameter and path
            const url = new URL(href, window.location.origin);
            const pageParam = url.searchParams.get('page');
            
            // Clone current URL parameters
            const newParams = new URLSearchParams(window.location.search);
            
            // Update page parameter
            if (pageParam) {
                newParams.set('page', pageParam);
            }
            
            // Build new URL with all parameters
            const newHref = url.pathname + '?' + newParams.toString();
            link.setAttribute('href', newHref);
        });
    }
});

// Helper function to format dates nicely
function formatDateNicely(dateStr) {
    try {
        // Handle different formats
        let date;
        
        // Try parsing different date formats
        if (dateStr.match(/^\d{4}-\d{2}-\d{2}T/)) {
            // ISO format: 2025-03-03T10:01:18
            date = new Date(dateStr);
        } 
        else if (dateStr.match(/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}/)) {
            // Format: 2025-03-03 10:01:18
            date = new Date(dateStr.replace(' ', 'T'));
        }
        else if (dateStr.match(/^\d{4}-\d{2}-\d{2}$/)) {
            // Format: 2025-03-03
            date = new Date(dateStr);
        }
        else {
            // Unknown format, return as is
            return dateStr;
        }
        
        // If date parsing failed, return original
        if (isNaN(date.getTime())) {
            return dateStr;
        }
        
        // Format the date as day number and time 
        const day = date.getDate();
        const time = date.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
        
        return `${day}, ${time}`;
    } catch (e) {
        console.error("Error formatting date:", e);
        return dateStr;
    }
}
