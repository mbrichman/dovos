    <script>
        // Highlight the search term in results (optional enhancement)
        document.addEventListener('DOMContentLoaded', function() {
            const searchInput = document.querySelector('input[name="query"]');
            if (searchInput && searchInput.value) {
                const searchTerm = searchInput.value.trim();
                if (searchTerm) {
                    const resultBodies = document.querySelectorAll('.result-body');
                    
                    resultBodies.forEach(body => {
                        // Skip highlighting within tags
                        const content = body.innerHTML;
                        const regex = new RegExp(`(${searchTerm})`, 'gi');
                        
                        // This is a simple approach - for production, you'd want a more robust HTML-aware solution
                        body.innerHTML = content.replace(regex, '<mark>$1</mark>');
                    });
                }
            }
        });
    </script>
