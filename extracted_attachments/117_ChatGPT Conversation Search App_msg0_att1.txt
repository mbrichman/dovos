        document.addEventListener('DOMContentLoaded', function() {
            // Format dates
            formatDates();
            
            // Set up filtering
            setupFilters();

        function formatDates() {
            const dateElements = document.querySelectorAll('.date-to-format');
            
            dateElements.forEach(function(element) {
                const dateStr = element.textContent.trim();
                const formattedDate = formatDateNicely(dateStr);
                element.textContent = formattedDate;
            });
        }
        
        function setupFilters() {
            const sourceFilter = document.getElementById('sourceFilter');
            const dateFilter = document.getElementById('dateFilter');
            const sortOrder = document.getElementById('sortOrder');
            
            if (!sourceFilter || !dateFilter || !sortOrder) return;
            
            const filters = [sourceFilter, dateFilter, sortOrder];
            
            filters.forEach(filter => {
                filter.addEventListener('change', applyFilters);
            });
            
            // Apply filters on load (in case of page refresh with selection)
            applyFilters();
        }
        
        function applyFilters() {
            const sourceFilter = document.getElementById('sourceFilter').value;
            const dateFilter = document.getElementById('dateFilter').value;
            const sortOrder = document.getElementById('sortOrder').value;
            
            // Get all conversation cards
            const cards = document.querySelectorAll('.conversation-card');
            
            cards.forEach(card => {
                // Start with visible
                let shouldShow = true;
                
                // Apply source filter
                if (sourceFilter !== 'all') {
                    const cardSource = card.getAttribute('data-source');
                    if (cardSource !== sourceFilter) {
                        shouldShow = false;
                    }
                }
                
                // Apply date filter
                if (shouldShow && dateFilter !== 'all') {
                    const cardDate = new Date(card.getAttribute('data-date'));
                    const now = new Date();
                    
                    if (isNaN(cardDate.getTime())) {
                        // If date parsing fails, show the card anyway
                    } else {
                        // Apply date filters
                        if (dateFilter === 'today') {
                            if (cardDate.toDateString() !== now.toDateString()) {
                                shouldShow = false;
                            }
                        } else if (dateFilter === 'week') {
                            const weekAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
                            if (cardDate < weekAgo) {
                                shouldShow = false;
                            }
                        } else if (dateFilter === 'month') {
                            const monthAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);
                            if (cardDate < monthAgo) {
                                shouldShow = false;
                            }
                        } else if (dateFilter === 'year') {
                            const yearAgo = new Date(now.getTime() - 365 * 24 * 60 * 60 * 1000);
                            if (cardDate < yearAgo) {
                                shouldShow = false;
                            }
                        }
                    }
                }
                
                // Show or hide the card
                card.style.display = shouldShow ? 'block' : 'none';
            });
            
            // Apply sort order - Note: This would require DOM manipulation
            // For better performance, this should be done server-side by changing the sort parameter in the URL
            
            // Show or hide date headers based on visible cards
            updateDateHeaders();
        }
        
        function updateDateHeaders() {
            // Get all date headers
            const dateHeaders = document.querySelectorAll('.date-header');
            
            dateHeaders.forEach(header => {
                // Find the next header or the end
                let nextHeader = header.nextElementSibling;
                while (nextHeader && !nextHeader.classList.contains('date-header')) {
                    nextHeader = nextHeader.nextElementSibling;
                }
                
                // Check if there are any visible cards between this header and the next
                let hasVisibleCards = false;
                let current = header.nextElementSibling;
                
                while (current && current !== nextHeader) {
                    if (current.classList.contains('conversation-card') && current.style.display !== 'none') {
                        hasVisibleCards = true;
                        break;
                    }
                    current = current.nextElementSibling;
                }
                
                // Show or hide the header
                header.style.display = hasVisibleCards ? 'block' : 'none';
            });
        }

function formatDateNicely(dateStr) {
    try {
        // Handle different formats
        let date;
        
        // Try parsing different date formats
        if (dateStr.match(/^\d{4}-\d{2}-\d{2}T/)) {
            // ISO format: 2025-03-03T10:01:18
            date = new Date(dateStr);
        } 
        else if (dateStr.match(/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}/)) {
            // Format: 2025-03-03 10:01:18
            date = new Date(dateStr.replace(' ', 'T'));
        }
        else if (dateStr.match(/^\d{4}-\d{2}-\d{2}$/)) {
            // Format: 2025-03-03
            date = new Date(dateStr);
        }
        else {
            // Unknown format, return as is
            return dateStr;
        }
        
        // If date parsing failed, return original
        if (isNaN(date.getTime())) {
            return dateStr;
        }
        
        // Format the date as day number and time 
        const day = date.getDate();
        const time = date.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
        
        return `${day}, ${time}`;
    } catch (e) {
        console.error("Error formatting date:", e);
        return dateStr;
    }
}
